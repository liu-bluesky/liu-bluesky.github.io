# js 代码

data(){
	return {
			list: [],
				triggered: false,
				_freshing: false // 是否开启下拉刷新
	}
},
methods: {
init() {
				this.formFilter = {}
				this.pageNum = 1
				this.list = []
				this.getData()
			},
			//获取数据
			getData(data) {
				let params = Object.assign({
						searchType: this.searchType,
						pageNum: this.pageNum,
						pageSize: 10,
						name: this.name
					},
					data
				)
				getCustomerList(params)
					.then(response => {
						if (this.pageNum <= 1) {
							this.list = response.data.list;
						} else {
							this.list = this.list.concat(response.data.list)
						}
						this.triggered = false; // 需要重置
						this._freshing = false;
					}).catch(err => {
						this.triggered = false; // 需要重置
						this._freshing = false;
					})

			},
			//自定义下拉刷新被触发
			onRefresh() {
				if (this._freshing) return;
				this.triggered = true; // 需要重置
				this._freshing = true;

				this.init()
			},
			//自定义下拉刷新被复位
			onRestore() {
				this.triggered = 'restore'; // 需要重置
			},
			//触底 触发事件
			scrolltolower() {
				this.pageNum += 1
				this.getData()
			},
}
