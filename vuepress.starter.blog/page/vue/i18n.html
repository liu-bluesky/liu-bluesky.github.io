创建 /@ts/language/index.ts 文件夹

import {createI18n} from 'vue-i18n';
import { App } from 'vue';
import zh from '/@ts/language/lang/zh';
import en from '/@ts/language/lang/en';
export const i18n = createI18n({
    locale: 'zh',//zh 中文
    messages: {
      'zh':zh,
      'en': en
    }
  });
/**
 * 挂载到vue实例函数
 * @param app vue实例
 */
export function setupI18n(app: App<Element>) {
    app.use(i18n);
  }
