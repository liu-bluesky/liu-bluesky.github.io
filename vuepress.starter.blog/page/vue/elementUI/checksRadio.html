# vue3.0 版本 element plus

<template>
  <div>
    <el-input v-model="values" v-if="DIY" @focus="show"></el-input>
    <el-dialog
      :close-on-click-modal="false"
      :title="title"
      :visible.sync="dialogVisible"
      append-to-body
      width="900px"
      height="80%"
    >
      <SearchForm :formItems="formItems" @change="handleSearch"></SearchForm>
      <el-table
        v-loading="loading"
        @select-all="handleSelectionChange"
        ref="tableList"
        @selection-change="handleSelectionChange"
        :data="tableList"
        class="mt20"
      >
        <el-table-column v-if="multiple" type="selection" width="55">
        </el-table-column>
        <el-table-column v-if="!multiple" width="55">
          <template slot-scope="scope">
            <el-radio v-model="currentItem" :label="scope.row"
              >{{ '' }}</el-radio
            >
          </template>
        </el-table-column>
        <el-table-column
          label="ID"
          align="center"
          prop="id"
          :show-overflow-tooltip="true"
        />
        <el-table-column
          label="设备名称"
          align="center"
          prop="name"
          :show-overflow-tooltip="true"
        />
        <el-table-column
          label="产品名称"
          align="center"
          prop="productName"
          :show-overflow-tooltip="true"
        />
        <el-table-column
          label="注册时间"
          align="center"
          prop="registryTime"
          :show-overflow-tooltip="true"
        >
          <template slot-scope="scope">
            {{ formatDate(scope.row.registryTime) }}
          </template>
        </el-table-column>
        <el-table-column
          label="状态"
          align="center"
          prop="state.text"
          :show-overflow-tooltip="true"
        ></el-table-column>
      </el-table>

      <pagination
        :total="total"
        :page.sync="queryParams.pageIndex"
        :limit.sync="queryParams.pageSize"
        @pagination="getList"
      />
      <span slot="footer" class="dialog-footer">
        <el-button @click="cancel">取 消</el-button>
        <el-button type="primary" @click="handleSave">保存</el-button>
      </span>
    </el-dialog>
  </div>
</template>

<script>
  import { deviceInstanceList } from "@/api/instance";

  export default {
    props: {
      index: {
        type: [Array, String, Number],
        default: 0,
      },
      title: {
        type: [Array, String, Number],
        default: "选择用户",
      },
      value: {
        type: [Array, String, Number],
        default: "",
      },
      multiple: {
        type: Boolean,
        default: true,
      },
      selectId: {
        type: [Array, String, Number, Object],
        default: "",
      },
      type: {
        type: [Array, String, Number, Object],
        default: "id$in-dimension$org$not",
      },
      DIY: {
        type: Boolean,
        default: false,
      },
    },
    mode: {
      prop: "value",
      event: "change",
    },
    data() {
      return {
        currentItem: {},
        values: "",
        dialogVisible: false,
        total: 0,
        loading: false,
        tableList: [],
        currentSearch: {},
        queryParams: {
          pageSize: 10,
          pageIndex: 1,
        },
        selection: [],
        formItems: [
          {
            placeholder: "请输入设备ID",
            key: "id$LIKE",
            type: "string",
          },

          {
            placeholder: "请输入设备名称",
            key: "name$LIKE",
            type: "string",
          },
        ],
      };
    },
    methods: {
      handleSearch(data) {
        this.currentSearch = data;
        this.queryParams.pageIndex = 1;
        this.getData();
      },
      cancel() {
        this.dialogVisible = false;
        this.selection = [];
        this.$refs.tableList.clearSelection();
      },
      handleSave() {
        if (this.multiple) {
          let ids = this.selection.map((res) => res.id).join(",");
          this.$emit("change", ids, {
            selection: this.selection,
            index: this.index,
          });
          this.$emit("save", this.selection);
          this.values = this.selection.map((res) => res.name).join(",");
          this.selection = [];
          this.dialogVisible = false;
        } else {
          if (this.currentItem.id) {
            this.$emit("change", this.currentItem.id, {
              currentItem: this.currentItem,
              index: this.index,
            });

            this.$emit("save", this.currentItem);
            this.dialogVisible = false;
          } else {
            this.$DIYMessage.error("请选择设备");
          }
        }
        this.$refs.tableList.clearSelection();
      },
      handleSelectionChange(rows) {
        this.selection = rows;
      },
      show() {
        this.dialogVisible = true;
        this.getList();
      },
      getList() {
        this.loading = true;
        deviceInstanceList({
          ...this.currentSearch,
          ...this.$encodeQueryParam({
            sorts: {
              order: "desc",
              field: "registryTime",
            },
          }),
          ...this.queryParams,
          pageIndex: this.queryParams.pageIndex - 1,
        })
          .then((res) => {
            this.loading = false;
            this.tableList = res.result.data;
            this.total = res.result.total;
          })
          .catch((err) => {
            this.loading = false;
          });
      },
    },
  };
</script>
